precision mediump float;
varying vec2 textureCoordinate;
uniform sampler2D inputImageTexture;
uniform float alpha;

void main()
{
	gl_FragColor = texture2D(inputImageTexture, textureCoordinate);
	gl_FragColor.a *= alpha;
}
